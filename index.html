// Privacy Policy Detection Module
class PrivacyDetector {
  constructor() {
    this.patterns = [
      /privacy\s+policy/i,
      /terms\s+(of\s+service|and\s+conditions)/i,
      /data\s+(collection|usage|processing)/i
    ];
    this.confidence = 0;
    console.log("Privacy Detector initialized");
  }

  async scanPage() {
    const bodyText = document.body.innerText;
    const url = window.location.href;
    
    // Check URL patterns
    if (url.includes("privacy") || 
        url.includes("terms")) {
      this.confidence += 0.4;
    }
    
    // Check content patterns
    for (const pattern of this.patterns) {
      if (pattern.test(bodyText)) {
        this.confidence += 0.2;
        console.log(`Matched pattern: ${pattern}`);
      }
    }
    
    if (this.confidence > 0.5) {
      return this.extractPolicyText();
    }
    
    return null;
  }
  
  extractPolicyText() {
    // Extract main content
    const mainContent = this.findMainContentElement();
    return {
      text: mainContent.innerText,
      html: mainContent.innerHTML,
      confidence: this.confidence
    };
  }
}

// Initialize and run detector
const detector = new PrivacyDetector();
detector.scanPage().then(result => {
  if (result) {
    chrome.runtime.sendMessage({
      action: "POLICY_DETECTED",
      payload: result
    });
  }
});
                            

// Privacy Risk Analysis Module
class PrivacyAnalyzer {
  constructor() {
    this.riskTerms = {
      high: [
        "sell your data", "share with third parties",
        "targeted advertising", "location tracking"
      ],
      medium: [
        "collect personal information", "store your data",
        "cookies", "behavioral analysis"
      ],
      low: [
        "anonymous statistics", "aggregate data",
        "improve service", "troubleshooting"
      ]
    };
  }

  async analyzePolicy(policyText) {
    const sentences = this.splitIntoSentences(policyText);
    const risks = {
      high: [],
      medium: [],
      low: []
    };
    
    // Analyze each sentence for risk terms
    for (const sentence of sentences) {
      const sentenceLower = sentence.toLowerCase();
      
      // Check for high risks
      for (const term of this.riskTerms.high) {
        if (sentenceLower.includes(term)) {
          risks.high.push({
            sentence,
            term,
            index: sentenceLower.indexOf(term)
          });
        }
      }
      
      // Similar checks for medium and low risks...
    }
    
    // Calculate overall risk score
    const riskScore = this.calculateRiskScore(risks);
    
    return {
      risks,
      riskScore,
      riskLevel: this.getRiskLevel(riskScore)
    };
  }
  
  splitIntoSentences(text) {
    return text.split(/[.!?]+/)
      .map(s => s.trim())
      .filter(s => s.length > 10);
  }
  
  calculateRiskScore(risks) {
    return (risks.high.length * 10) + 
           (risks.medium.length * 3) + 
           (risks.low.length * 1);
  }
}
                            

// OpenAI Summary Generator Module
class SummaryGenerator {
  constructor(apiKey) {
    this.apiKey = apiKey;
    this.endpoint = "https://api.openai.com/v1/chat/completions";
  }

  async generateSummary(policyText, risks) {
    const prompt = this.createPrompt(policyText, risks);
    
    try {
      const response = await fetch(this.endpoint, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "Authorization": `Bearer ${this.apiKey}`
        },
        body: JSON.stringify({
          model: "gpt-3.5-turbo",
          messages: [{
            role: "system",
            content: "You analyze privacy policies and summarize key points of concern in clear language."
          }, {
            role: "user",
            content: prompt
          }],
          max_tokens: 150,
          temperature: 0.7
        })
      });
      
      const data = await response.json();
      return data.choices[0].message.content;
    } catch (error) {
      console.error("Error generating summary:", error);
      return "Unable to generate summary at this time.";
    }
  }
  
  createPrompt(policyText, risks) {
    return `Please analyze this privacy policy and provide a concise summary 
    focusing on concerns regarding data collection, sharing, and user privacy.
    
    Policy text: "${policyText.substring(0, 1000)}..."
    
    Identified risks: ${JSON.stringify(risks)}
    
    Provide a brief summary (2-3 sentences) highlighting the most concerning aspects.`;
  }
}

New EU AI Regulation Imposes Strict Privacy Requirements

Elena Bergson Technology Policy Reporter
European Parliament in Brussels where the legislation was passed

In a landmark decision that will reshape the landscape of artificial intelligence development and deployment within its borders, the European Union has approved the comprehensive AI Act, setting unprecedented standards for privacy protection in AI systems.

The regulation, which passed with a vote of 548-78, establishes a tiered approach to AI regulation based on risk levels, with the strictest rules applying to systems that interact with personal data or make automated decisions affecting individuals.

Key Privacy Provisions

  • Data Minimization Requirements: AI systems must be designed to use the minimum amount of personal data necessary to function effectively.
  • Transparency Obligations: Users must be clearly informed when interacting with AI systems, with explicit disclosure of data collection practices.
  • Right to Explanation: Individuals have the right to understand how automated decisions affecting them are made.
  • Prohibition of Real-time Biometric Identification: Strict limitations on facial recognition and other biometric identification systems in public spaces.

The legislation imposes fines of up to 6% of global annual revenue for serious violations, significantly higher than the GDPR's maximum penalties.

"This regulation sets a new global standard for responsible AI that respects fundamental rights while fostering innovation," said Dragos Tudorache, co-rapporteur of the legislation. "It ensures that AI serves people, not the other way around."

Global Impact

Industry experts anticipate a "Brussels Effect" similar to what occurred with GDPR, where global technology companies adapt their worldwide practices to comply with EU regulations rather than maintaining different standards for different markets.

The regulation is expected to come into force in early 2024, with a phased implementation period giving companies time to adapt their systems to the new requirements.
